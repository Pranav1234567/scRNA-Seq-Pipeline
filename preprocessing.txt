#!/bin/bash

echo "Welcome "$USER", to this BASH script on preprocessing scRNA-Seq Data"

if [ -z $1 ]  || [ -z $2 ] || [ -z $3 ] || [ -z $4 ] 
then 
	echo "must type './preprocessing.txt </path/to/project> </path/to/Data> </path/to/SRAToolkit> </path/to/FastQC>' "
else
	#Predefined by user
	BASE=$1

	PROJ=${BASE}/Project

	ANALYSIS=${PROJ}/Analysis

	RAW=${ANALYSIS}/Raw
	TRIM = ${ANALYSIS}/Trim
	QC=${ANALYSIS}/QC
	INFO=${ANALYSIS}/INFO

	mkdir ${PROJ} ${ANALYSIS} ${TRIM} ${RAW} ${QC} ${INFO}

	#Once directories are created, we can take in the input data, convert to FASTQ and generate quality reports
        	rawname=`echo $2 | cut -f1 -d'.'`
		echo "--------------------------------------"
        	echo "Making two FASTQ files from SRA..."
        	echo "----------------------------------"
                cd $3
               	./fastq-dump -O ${RAW} -I --split-files $2
                echo "----------------------------------"
                echo "DONE"
                echo "------------------------------------"
                echo "generating quality control reports using FASTQC"
		echo "------------------------------------"
                $4 --threads 8 ${rawname}_1.fastq -o ${QC}
                echo "-----------------------------------------"
                $4 --threads 8 ${rawname}_2.fastq -o ${QC}
                echo "DONE"
	
	#Trimming

 		#Demultiplex
			# ./extractBarcodes.txt <fastq file containing barcode info> ${INFO} -> where the barcode file will be saved   
			cd $3
		#	./fastq-multx <barcode file> <input> -o <output '%'>
		#Adapters	
			./removeAdapters.txt 				
fi
