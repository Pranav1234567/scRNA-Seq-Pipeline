#!/bin/bash

echo "Welcome "$USER", to this BASH script on preprocessing scRNA-Seq Data"

if [ -z $1 ]  || [ -z $2 ] || [ -z $3 ] || [ -z $4 ] 
then 
	echo "must type './preprocessing.txt </path/to/project> </path/to/Data> </path/to/SRAToolkit> </path/to/FastQC>' "
else
	#Predefined by user
	BASE=$1

	PROJ=${BASE}/Project

	ANALYSIS=${PROJ}/Analysis

	RAW=${ANALYSIS}/Raw
	QC=${ANALYSIS}/QC
	INFO=${ANALYSIS}/Info
	DEMUX=${ANALYSIS}/Demux
	SCRIPTS="Documents/scripts"

	mkdir ${PROJ} ${ANALYSIS} ${RAW} ${QC} ${INFO} ${DEMUX}

	#Once directories are created, we can take in the input data, convert to FASTQ and generate quality reports
        	rawname=`echo $2 | cut -f1 -d'.'`
		echo "--------------------------------------"
        	echo "Making two FASTQ files from SRA..."
        	echo "----------------------------------"
                cd $3
               	./fastq-dump -O ${RAW} -I --split-files $2
                echo "----------------------------------"
                echo "DONE"
                echo "------------------------------------"
                echo "generating quality control reports using FASTQC"
		echo "------------------------------------"
                $4 --threads 8 ${rawname}_1.fastq -o ${QC}
                echo "-----------------------------------------"
                $4 --threads 8 ${rawname}_2.fastq -o ${QC}
                echo "DONE"

 	#Demultiplex
		cd 
		cd ${SCRIPTS}
		chmod 777 extractBarcodes.txt

		for f in $(ls ${RAW}/SRR*_1.fastq)
		do	
			./extractBarcodes.txt ${f} ${INFO}
		done   
		
		for f in $(ls ${RAW}/SRR*_1.fastq)
		do
			for g in $(ls ${RAW}/SRR*_2.fastq)
			do
				mkdir ${DEMUX}/$(basename ${rawname}_1)/
				mkdir ${DEMUX}/$(basename ${rawname}_2)/
				fastq-multx ${INFO}/barcodes.tab ${f} ${g} -o ${DEMUX}/$(basename ${rawname}_1)/cell.%.fq ${DEMUX}/$(basename ${rawname}_2)/cell.%.fq
			done
		done
		 
	#Trimming Adapters
		chmod 777 removeAdapters.txt
		./removeAdapters.txt ${DEMUX} ${rawname} /Users/Pranav/Documents/Research/AnalysisResults/adapters.fa 				
fi
